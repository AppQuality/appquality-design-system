import { Meta } from "@storybook/addon-docs/blocks";
import * as colors from "../shared/colors";

export const ColorBlock = ({ colors, color }) => {
  const size = "180px";
  return (
    <div style={{ float: "left", margin: "5px 0" }}>
      {color}: {colors[color]}
      <div
        style={{
          backgroundColor: colors[color],
          width: size,
          height: size,
          border: "1px solid",
        }}
      ></div>
    </div>
  );
};

export const myColors = Object.fromEntries(Object.entries(colors));

<Meta title="Colors" />

<>
  {Object.keys(myColors).map((c) => {
    return <ColorBlock colors={myColors} color={c} />;
  })}
</>
